```shell
git clone repo_url
```

```shell
cd project_name && yarn
```

## mysql

```shell
mysql -u root -p
```

```shell
CREATE USER 'username'@'localhost' IDENTIFIED BY 'password';
```

```shell
CREATE DATABASE db_name;
```

```shell
USE db_name;
```

```shell
GRANT ALL PRIVILEGES ON *.* TO 'username'@'localhost';
```
